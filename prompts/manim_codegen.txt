EXAMPLE 1 — equation_transform: softmax derivation

```python
from manim import *

class SoftmaxDerivationScene(Scene):
    def construct(self):
        title = Text("Softmax Function", font_size=40).to_edge(UP)
        self.play(FadeIn(title))
        self.wait(0.5)

        eq1 = MathTex(r"\text{softmax}(z_i) = \frac{e^{z_i}}{\sum_j e^{z_j}}")
        self.play(Write(eq1))
        self.wait(1.5)

        eq2 = MathTex(r"\text{softmax}(z_i) = \frac{e^{z_i - \max(z)}}{\sum_j e^{z_j - \max(z)}}")
        label = Text("(numerically stable form)", font_size=22, color=YELLOW).next_to(eq2, DOWN)
        self.play(TransformMatchingTex(eq1, eq2))
        self.play(FadeIn(label))
        self.wait(2)

        self.play(FadeOut(eq2), FadeOut(label))
        note = Text("Outputs sum to 1 and are all positive.", font_size=28).move_to(ORIGIN)
        self.play(Write(note))
        self.wait(2)
```

EXAMPLE 2 — flow: transformer encoder pipeline

```python
from manim import *

class TransformerEncoderScene(Scene):
    def construct(self):
        title = Text("Transformer Encoder Block", font_size=36).to_edge(UP)
        self.play(FadeIn(title))

        stages = ["Input Embedding", "Multi-Head Attention", "Add & Norm", "Feed Forward", "Add & Norm"]
        colors = [BLUE, GREEN, TEAL, ORANGE, TEAL]
        boxes = []
        start_y = 1.8

        for i, (stage, color) in enumerate(zip(stages, colors)):
            rect = RoundedRectangle(width=3.5, height=0.6, corner_radius=0.1, color=color)
            text = Text(stage, font_size=22).move_to(rect.get_center())
            box = VGroup(rect, text).move_to([0, start_y - i * 1.0, 0])
            boxes.append(box)
            self.play(FadeIn(box), run_time=0.4)

        self.wait(0.5)
        for i in range(len(boxes) - 1):
            arrow = Arrow(boxes[i].get_bottom(), boxes[i+1].get_top(), buff=0.05)
            self.play(Create(arrow), run_time=0.3)

        self.wait(2)
```

EXAMPLE 3 — geometric: attention as weighted sum of values

```python
from manim import *
import numpy as np

class AttentionWeightedSumScene(Scene):
    def construct(self):
        title = Text("Attention: Weighted Sum of Values", font_size=34).to_edge(UP)
        self.play(FadeIn(title))
        self.wait(0.3)

        # Three value vectors as vertical bars
        values = [("V₁", BLUE, LEFT * 3), ("V₂", GREEN, ORIGIN), ("V₃", RED, RIGHT * 3)]
        weights = [0.6, 0.3, 0.1]
        bars = []

        for (label, color, pos), w in zip(values, weights):
            bar = Rectangle(width=0.8, height=w * 3, color=color, fill_opacity=0.7)
            bar.move_to(pos + DOWN * 0.5)
            lbl = Text(label, font_size=26).next_to(bar, DOWN)
            w_lbl = Text(f"α={w}", font_size=22, color=YELLOW).next_to(bar, UP)
            bars.append(VGroup(bar, lbl, w_lbl))
            self.play(FadeIn(bars[-1]), run_time=0.6)

        self.wait(1)
        result = MathTex(r"\text{Output} = \sum_i \alpha_i V_i").to_edge(DOWN, buff=1)
        self.play(Write(result))
        self.wait(2)
```
