```python
from manim import *
import numpy as np

class SelfAttentionScene(Scene):
    def construct(self):
        title = Text("Self-Attention  ·  Attention Is All You Need", font_size=32).to_edge(UP)
        self.play(FadeIn(title))
        self.wait(0.5)

        # Beat 1: token row
        tokens = ["The", "cat", "sat", "on", "mat"]
        circles, labels = node_row(tokens, [BLUE_C]*5, center=CONTENT_CENTER + UP*1.85)
        self.play(LaggedStart(*[Create(c) for c in circles], lag_ratio=0.12))
        self.play(LaggedStart(*[FadeIn(l) for l in labels], lag_ratio=0.12))
        self.wait(1.0)

        # Beat 2: attention weights from "sat"
        weights = [0.03, 0.14, 0.68, 0.09, 0.06]
        q_ring = Circle(radius=0.44, color=YELLOW, stroke_width=3).move_to(circles[2])
        self.play(Create(q_ring))
        w_objs = VGroup()
        for i, w in enumerate(weights):
            col = interpolate_color(GREY_C, YELLOW, min(1.0, w*2.6))
            ln = Line(circles[2].get_center(), circles[i].get_center(),
                      stroke_width=1.2+w*15, color=col, stroke_opacity=0.85)
            wt = Text(f"{w:.2f}", font_size=14, color=col).next_to(circles[i], DOWN, buff=0.14)
            w_objs.add(ln, wt)
        self.play(LaggedStart(*[Create(o) for o in w_objs[::2]], lag_ratio=0.1))
        self.play(LaggedStart(*[FadeIn(o) for o in w_objs[1::2]], lag_ratio=0.1))
        self.wait(2.0)
        self.play(FadeOut(w_objs), FadeOut(q_ring), FadeOut(circles), FadeOut(labels))

        # Beat 3: full attention heatmap
        attn = [[0.55,0.20,0.12,0.07,0.06],[0.18,0.48,0.22,0.07,0.05],
                [0.03,0.14,0.68,0.09,0.06],[0.05,0.10,0.28,0.45,0.12],
                [0.04,0.07,0.18,0.24,0.47]]
        heat = heatmap(attn, row_labels=tokens, col_labels=tokens,
                       cell_size=0.62, center=CONTENT_CENTER + DOWN*0.55)
        self.play(FadeIn(heat), run_time=0.9)
        note = Text("'sat' attends most to itself (0.68)", font_size=17, color=YELLOW)
        note.next_to(heat, DOWN, buff=0.28)
        self.play(FadeIn(note))
        self.wait(2.5)
```
